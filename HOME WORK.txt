        // Use string concatenation to combine two strings in a Console.WriteLine statement
        Console.WriteLine("My name is " + myName);

        // Perform a simple arithmetic operation (5 + 3) and print the result
        int result = 5 + 3;
        Console.WriteLine("Result of 5 + 3 is: " + result);

        // Print two additional lines of text
        Console.WriteLine("This is another line of text.");
        Console.WriteLine("Programming is fun!");

        // Print out your favorite number
        int favoriteNumber = 42; // Change this to your actual favorite number
        Console.WriteLine("My favorite number is: " + favoriteNumber);

        // Display today's date

        // Read user input and print it out
        Console.Write("Enter something: ");
        string userInput = Console.ReadLine();
        Console.WriteLine("You entered: " + userInput);
    }
}
